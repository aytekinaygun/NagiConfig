### VIRTUALENV KURULUM
# En yeni sürümü yükle
apt-get install python3.x

# pip yükle
apt-get install python3-pip

# virtualenv kur
pip3 install virtualenv

# proje dizini oluştur
mkdir ...../NagiConfig
cd ....../NigiConfig

# dizinin içindeyken (python3 kullanmak için)
virtualenv -p python3 my_env

# çalıştır
source ./my_env/bin/activate

# durdur
deactivate
